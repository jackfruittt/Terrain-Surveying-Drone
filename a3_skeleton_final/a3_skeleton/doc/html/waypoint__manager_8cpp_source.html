25bool WaypointManager::isTraversable(const geometry_msgs::msg::Pose& start, const geometry_msgs::msg::Pose& end) {
26 // Get the elevations at the start and end points
27 double start_elevation = getElevationAtPoint_(start.position);
28 double end_elevation = getElevationAtPoint_(end.position);
29
30 // Calculate horizontal distance
31 double dx = end.position.x - start.position.x;
32 double dy = end.position.y - start.position.y;
33 double horizontal_distance = std::sqrt(dx*dx + dy*dy);
34
35 if (horizontal_distance < 0.001) {
36 // Points are too close horizontally, consider it traversable
37 return true;
38 }
39
40 // Calculate the elevation change
41 double elevation_change = std::abs(end_elevation - start_elevation);
42
43 // Calculate gradient as percentage
44 double gradient = (elevation_change / horizontal_distance) * 100.0;
45
46 RCLCPP_DEBUG(logger_, "Path from (%.2f,%.2f) to (%.2f,%.2f): gradient=%.2f%% (max=%.2f%%)",
47 start.position.x, start.position.y,
48 end.position.x, end.position.y,
49 gradient, max_gradient_);
50
51 // Check intermediate points along the path for more accuracy
52 int num_checks = static_cast<int>(horizontal_distance / 0.5) + 1; // Check every 0.5m
53 for (int i = 1; i < num_checks; i++) {
54 double t1 = static_cast<double>(i-1) / num_checks;
55 double t2 = static_cast<double>(i) / num_checks;
56
57 // Interpolate positions
58 geometry_msgs::msg::Point p1, p2;
59 p1.x = start.position.x + t1 * dx;
60 p1.y = start.position.y + t1 * dy;
61 p2.x = start.position.x + t2 * dx;
62 p2.y = start.position.y + t2 * dy;
63
64 // Get elevations
65 double elev1 = getElevationAtPoint_(p1);
66 double elev2 = getElevationAtPoint_(p2);
67
68 // Calculate segment distance and gradient
69 double segment_dist = std::sqrt(std::pow(p2.x - p1.x, 2) + std::pow(p2.y - p1.y, 2));
70 double segment_gradient = (std::abs(elev2 - elev1) / segment_dist) * 100.0;
71
72 if (segment_gradient > max_gradient_) {
73 RCLCPP_WARN(logger_, "Segment gradient %.2f%% exceeds max %.2f%%",
74 segment_gradient, max_gradient_);
75 return false;
76 }
77 }
78
79 return gradient <= max_gradient_;
80}

82std::vector<geometry_msgs::msg::Pose> WaypointManager::generateGroundWaypoints(
83 const geometry_msgs::msg::Pose& start,
84 const geometry_msgs::msg::Pose& end,
85 double spacing)
86{
87 std::vector<geometry_msgs::msg::Pose> waypoints;
88
89 // Calculate horizontal distance between start and end
90 double dx = end.position.x - start.position.x;
91 double dy = end.position.y - start.position.y;
92 double horizontal_distance = std::sqrt(dx*dx + dy*dy);
93
94 // Calculate number of segments
95 int num_segments = std::max(1, static_cast<int>(horizontal_distance / spacing));
96
97 // Create waypoints
98 for (int i = 0; i <= num_segments; i++) {
99 geometry_msgs::msg::Pose waypoint;
100
101 // Interpolate x and y position
102 double t = static_cast<double>(i) / static_cast<double>(num_segments);
103 waypoint.position.x = start.position.x + t * dx;
104 waypoint.position.y = start.position.y + t * dy;
105
106 // Get ground elevation at this point
107 waypoint.position.z = getElevationAtPoint_(waypoint.position);
108
109 // Safety margin for drone clearance
110 waypoint.position.z += 2.0;
111
112 // Calculate orientation to next waypoint
113 if (i < num_segments) {
114 // Calculate yaw angle to next point
115 double next_t = static_cast<double>(i + 1) / static_cast<double>(num_segments);
116 double next_x = start.position.x + next_t * dx;
117 double next_y = start.position.y + next_t * dy;
118
119 // Calculate yaw angle
120 double yaw = std::atan2(next_y - waypoint.position.y,
121 next_x - waypoint.position.x);
122
123 // Convert to quaternion
124 tf2::Quaternion q;
125 q.setRPY(0, 0, yaw);
126 waypoint.orientation = tf2::toMsg(q);
127 } else {
128 // For the last point, use the end pose orientation
129 waypoint.orientation = end.orientation;
130 }
131
132 waypoints.push_back(waypoint);
133 }
134
135 return waypoints;
136}

138std::vector<geometry_msgs::msg::Pose> WaypointManager::generateDroneWaypoints(
139 const geometry_msgs::msg::Pose& start,
140 const geometry_msgs::msg::Pose& end,
141 double spacing)
142{
143 std::vector<geometry_msgs::msg::Pose> waypoints;
144
145 // Calculate 3D distance between start and end (including Z difference)
146 double dx = end.position.x - start.position.x;
147 double dy = end.position.y - start.position.y;
148 double dz = end.position.z - start.position.z;
149 double total_distance = std::sqrt(dx*dx + dy*dy + dz*dz);
150
151 // Calculate number of segments for 1m spacing
152 int num_segments = std::max(1, static_cast<int>(std::ceil(total_distance / spacing)));
153
154 RCLCPP_DEBUG(logger_, "Generating drone waypoints: distance=%.2fm, segments=%d",
155 total_distance, num_segments);
156
157 // Create intermediate waypoints every 1m along the 3D path
158 for (int i = 0; i <= num_segments; i++) {
159 geometry_msgs::msg::Pose waypoint;
160
161 // Interpolate position along the 3D line (including Z)
162 double t = static_cast<double>(i) / static_cast<double>(num_segments);
163 waypoint.position.x = start.position.x + t * dx;
164 waypoint.position.y = start.position.y + t * dy;
165 waypoint.position.z = start.position.z + t * dz; // Linear interpolation of altitude
166
167 // Calculate orientation toward the end goal
168 if (i < num_segments) {
169 // Calculate direction to end goal
170 double remaining_dx = end.position.x - waypoint.position.x;
171 double remaining_dy = end.position.y - waypoint.position.y;
172
173 // Calculate yaw angle
174 double yaw = std::atan2(remaining_dy, remaining_dx);
175
176 // Convert to quaternion
177 tf2::Quaternion q;
178 q.setRPY(0, 0, yaw);
179 waypoint.orientation = tf2::toMsg(q);
180 } else {
181 // For the last point, use the end pose orientation
182 waypoint.orientation = end.orientation;
183 }
184
185 waypoints.push_back(waypoint);
186 }
187
188 return waypoints;
189}

278
279 // Add all waypoints from this segment (excluding the last one to avoid duplicates)
280 for (size_t j = 0; j < segment_waypoints.size() - 1; j++) {
281 drone_waypoints.push_back(segment_waypoints[j]);
282 }
283 }
284

286 geometry_msgs::msg::Pose final_goal = goals.back();
287 if (final_goal.position.z <= 0.1) {
288 final_goal.position.z = 2.0; // Visualization at 2m for terrain following
289 }
290 drone_waypoints.push_back(final_goal);
291 }
292
293 // Add all waypoints to the message
294 for (const auto& waypoint : drone_waypoints) {
295 waypoints_msg.poses.push_back(waypoint);
296 }
297

299 waypoints_pub_->publish(waypoints_msg);
300
301 // Store the waypoints in the vehicle (for markers)
302 {
303 auto& non_const_vehicle = const_cast<VehicleData&>(vehicle);
304 std::lock_guard<std::mutex> lock(non_const_vehicle.goals_mutex);
305 non_const_vehicle.traversable_waypoints = drone_waypoints;
