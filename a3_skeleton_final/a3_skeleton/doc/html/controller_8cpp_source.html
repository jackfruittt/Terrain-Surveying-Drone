17bool Controller::processCommand(VehicleData &vehicle, const VehicleData::Command &command)

19 switch (command.type)
20 {

105bool Controller::takeoff(VehicleData &vehicle)
106{
107 RCLCPP_INFO(logger_, "Taking off %s...", vehicle.id.c_str());
108
109 // Publish takeoff command - send multiple times to ensure it's received
110 std_msgs::msg::Empty takeoff_msg;
111 for (int i = 0; i < 5; i++) // 5 repetitions
112 {
113 takeoff_pub_->publish(takeoff_msg);
114 std::this_thread::sleep_for(std::chrono::milliseconds(100)); // delay each message send to not overload
115 }
116
117 // Initial climb
118 current_altitude_adjustment_ = 1.0; // 1m/s, increase if needs to be more aggressive
119
120 // Log takeoff completion
121 RCLCPP_INFO(logger_, "Takeoff initiated for %s", vehicle.id.c_str());
122 vehicle.takeoff_complete = true;
123
124 // Send direct climb command
125 geometry_msgs::msg::Twist cmd_vel;
126 cmd_vel.linear.z = 1.0; // Force strong initial ascent
127 cmd_vel_pub_->publish(cmd_vel);
128
129 // Wait for drone to reach minimum safe height
130 rclcpp::Rate rate(10); // 10 Hz
131 int timeout = 100; // 10 seconds timeout
132
133 RCLCPP_INFO(logger_, "Waiting for drone to reach safe takeoff height...");
134
135 for (int i = 0; i < timeout; i++)
136 {
137 double altitude = 0.0;
138 {
139 std::lock_guard<std::mutex> lock(vehicle.odom_mutex);
140 altitude = vehicle.current_odom.pose.pose.position.z;
141 }
142
143 // Debug output at 1Hz
144 if (i % 10 == 0)
145 {
146 RCLCPP_INFO(logger_, "Takeoff progress: %.2fm (target: 2.5m)", altitude);
147 }
148
149 // Periodically resend climb command
150 if (i % 20 == 0)
151 {
152 cmd_vel.linear.z = 1.0;
153 cmd_vel_pub_->publish(cmd_vel);
154 }
155
156 if (altitude > 2.5)
157 {
158 RCLCPP_INFO(logger_, "Drone reached safe takeoff height: %.2fm", altitude);
159 break;
160 }
161
162 if (i == timeout - 1)
163 {
164 RCLCPP_WARN(logger_, "Takeoff timeout - drone reached %.2fm (wanted 2.5m)", altitude);
165 }
166
167 rate.sleep();
168 }
169
170 // Stabilise at takeoff height
171 cmd_vel.linear.z = 0.0;
172 cmd_vel_pub_->publish(cmd_vel);
173
174 // Reset altitude adjustment to neutral
175 current_altitude_adjustment_ = 0.0;
176

192bool Controller::flyToGoal(VehicleData& vehicle, const geometry_msgs::msg::Pose& goal) {
193 // Check if this specific goal has altitude override (z > 0.1), This makes it trigger for z > 2, idk why
194 bool use_manual_altitude = (goal.position.z > 0.1);
195
196 if (use_manual_altitude) {
197 RCLCPP_INFO(logger_, "Flying %s to goal: [%.2f, %.2f, %.2f] (MANUAL ALTITUDE)",
198 vehicle.id.c_str(), goal.position.x, goal.position.y, goal.position.z);
199 } else {
200 RCLCPP_INFO(logger_, "Flying %s to goal: [%.2f, %.2f, Z=terrain] (TERRAIN FOLLOWING)",
201 vehicle.id.c_str(), goal.position.x, goal.position.y);
202 }
203
204 rclcpp::Rate rate(10); // 10 Hz
205 int max_attempts = 1800; // Maximum 3 minutes
206
207 for (int i = 0; i < max_attempts; i++)
208 {
209 if (!vehicle.mission_active) {
210 return false;
211 }
212
213 // Get current position
214 geometry_msgs::msg::Point current_pos;
215 {
216 std::lock_guard<std::mutex> lock(vehicle.odom_mutex);
217 current_pos = vehicle.current_odom.pose.pose.position;
218 }
219
220 // Calculate horizontal distance to goal
221 double dx = goal.position.x - current_pos.x;
222 double dy = goal.position.y - current_pos.y;
223 double horizontal_distance = std::sqrt(dx*dx + dy*dy);
224
225 // Check if goal reached
226 bool goal_reached = false;
227 if (use_manual_altitude) {
228 double altitude_distance = std::abs(goal.position.z - current_pos.z);
229 goal_reached = (horizontal_distance < 0.5 && altitude_distance < 0.3);
230 } else {
231 goal_reached = (horizontal_distance < 0.5);
232 }
233
234 if (goal_reached) {
235 RCLCPP_INFO(logger_, "Goal reached for %s", vehicle.id.c_str());
236
237 // Brief hover then return
238 geometry_msgs::msg::Twist hover_cmd;
239 hover_cmd.linear.x = 0.0;
240 hover_cmd.linear.y = 0.0;
241
242 if (use_manual_altitude) {
243 double altitude_error = goal.position.z - current_pos.z;
244 hover_cmd.linear.z = altitude_error * 0.3;
245 hover_cmd.linear.z = std::clamp(hover_cmd.linear.z, -0.2, 0.2);
246 } else {
247 // Current altitude adjustment from controller thread
248 hover_cmd.linear.z = current_altitude_adjustment_;
249 }
250
251 cmd_vel_pub_->publish(hover_cmd);
252
253 // Short hover period then continue
254 for (int hover_count = 0; hover_count < 10; hover_count++) { // 1 second hover
255 cmd_vel_pub_->publish(hover_cmd);
256 rate.sleep();
257 }
258
259 return true;
260 }
261
262 // Gradual approach to avoid spiraling
263 geometry_msgs::msg::Twist cmd_vel;
264
265 // Horizontal movement with speed ramping
266 if (horizontal_distance > 0.1) {
267 double base_speed = 0.8; // Base speed in m/s
268
269 // Adaptive speed based on distance to avoid spiraling
270 double speed_factor;
271 if (horizontal_distance > 2.0) {
272 speed_factor = 1.0;
273 } else {
274 // Slow when approaching target
275 speed_factor = std::max(0.3, horizontal_distance / 2.0);
276 }
277
278 // Direct velocity
279 double target_speed = base_speed * speed_factor;
280 cmd_vel.linear.x = (dx / horizontal_distance) * target_speed;
281 cmd_vel.linear.y = (dy / horizontal_distance) * target_speed;
282
283 } else {
284 // Close to target - stop horizontal movement
285 cmd_vel.linear.x = 0.0;
286 cmd_vel.linear.y = 0.0;
287 }
288
289 // Altitude control
290 if (use_manual_altitude) {
291 double altitude_error = goal.position.z - current_pos.z;
292 cmd_vel.linear.z = altitude_error * 0.4;
293 cmd_vel.linear.z = std::clamp(cmd_vel.linear.z, -0.4, 0.4);
294 } else {
295 cmd_vel.linear.z = current_altitude_adjustment_;
296
297 if (current_pos.z > 8.0 && cmd_vel.linear.z > 0) {
298 cmd_vel.linear.z = std::min(cmd_vel.linear.z, 0.0);
299 }
300 }
301
302 cmd_vel_pub_->publish(cmd_vel);
303
304 // Less frequent debug output to reduce log spam
305 if (i % 20 == 0) { // Every 2 seconds
306 if (use_manual_altitude) {
307 RCLCPP_INFO(logger_, "Distance: %.2fm, altitude: %.2fm->%.2fm (MANUAL)",
308 horizontal_distance, current_pos.z, goal.position.z);
309 } else {
310 RCLCPP_INFO(logger_, "Distance: %.2fm, altitude: %.2fm (TERRAIN)",
311 horizontal_distance, current_pos.z);
312 }
313 }
314
315 rate.sleep();
316 }
317
318 RCLCPP_ERROR(logger_, "Failed to reach goal within time limit");
319 return false;
320}

365{
366 if (!laser_scan)
367 {
368 return false;
369 }
370
371 // Check ALL 360 degrees for obstacles
372 double critical_distance = 1.0; // 1m, emergency ascent
373 double warning_distance = 3.0; // 3m warning distance
374
375 bool obstacle_detected = false;
376 double closest_range = std::numeric_limits<double>::max();
377
378 // Check all laser readings
379 for (size_t i = 0; i < laser_scan->ranges.size(); i++)
380 {
381 float range = laser_scan->ranges[i];
382
383 // Check if reading is valid
384 if (!std::isnan(range) && !std::isinf(range) &&
385 range > laser_scan->range_min &&
386 range < laser_scan->range_max)
387 {
388
389 if (range < closest_range)
390 {
391 closest_range = range;
392 }
393
394 if (range < critical_distance)
395 {
396 obstacle_detected = true;
397
398 // Calculate angle of obstacle
399 float angle = laser_scan->angle_min + (i * laser_scan->angle_increment);
400 float angle_deg = angle * 180.0 / M_PI;
401
402 RCLCPP_ERROR(logger_, "OBSTACLE at %.2fm, angle %.1f degrees - EMERGENCY ASCENT!",
403 range, angle_deg);
404 }
405 else if (range < warning_distance)
406 {
407 // Simple throttling using static variable
408 static auto last_warning = std::chrono::steady_clock::now();
409 auto now = std::chrono::steady_clock::now();
410
411 if (std::chrono::duration_cast<std::chrono::milliseconds>(now - last_warning).count() > 1000)
412 {
413 float angle = laser_scan->angle_min + (i * laser_scan->angle_increment);
414 float angle_deg = angle * 180.0 / M_PI;
415
416 RCLCPP_WARN(logger_, "Obstacle at %.2fm, angle %.1f degrees",
417 range, angle_deg);
418 last_warning = now;
419 }
420 }
421 }
422 }
423
424 if (obstacle_detected)
425 {
426 RCLCPP_ERROR(logger_, "Closest obstacle at %.2fm - must ascend immediately!", closest_range);
427 }
428
429 return obstacle_detected;

432double Controller::calculateAltitudeAdjustment(double current_altitude, double sonar_range)
433{
434 // Print raw data for debugging
435 static int debug_counter = 0;
436 if (++debug_counter % 10 == 0)
437 {
438 RCLCPP_DEBUG(logger_, "RAW sonar=%.2fm, altitude=%.2fm", sonar_range, current_altitude);
439 }
440
441 // Strict validation - if sonar is reporting weird values, don't trust it
442 if (std::isnan(sonar_range) || std::isinf(sonar_range) ||
443 sonar_range <= 0.1 || sonar_range > 10.0)
444 {
445 RCLCPP_WARN(logger_, "Invalid sonar reading: %.2f - stopping adjustment", sonar_range);
446 return 0.0; // Return 0.0 to prevent drift
447 }
448
449 // Target height remains 2.0m
450 double target_height = 2.0;
451
452 // Calculate error - positive to go up
453 double error = target_height - sonar_range;
454
455 // Add integral term to eliminate steady-state error
456 static double integral_error = 0.0;
457 double dt = 0.1; // For 10Hz
458
459 // Only integrate when error is small to prevent windup
460 if (std::abs(error) < 0.5)
461 {
462 integral_error += error * dt;
463 // Clamp integral term to prevent excessive accumulation
464 integral_error = std::clamp(integral_error, -1.0, 1.0);
465 }
466 else
467 {
468 // Reset integral when error is large
469 integral_error = 0.0;
470 }
471
472 // Asymmetric PI control - different gains for ascent and descent
473 double kp, ki;
474 if (error > 0)
475 { // Too low - need to ascend
476 kp = 0.85; // Slightly reduced from 0.90 to prevent overshoot
477 ki = 0.15; // Add integral term for uphill
478 }
479 else
480 { // Too high - need to descend
481 kp = 0.45; // Slightly reduced from 0.50
482 ki = 0.10; // Less integral for downhill (gravity)
483 }
484
485 // Calculate adjustment with both P and I terms
486 double p_term = kp * error;
487 double i_term = ki * integral_error;
488 double adjustment = p_term + i_term;
489
490 // Allow faster ascent when too low (safety priority)
491 double max_ascent = (error > 1.0) ? 1.0 : 0.6; // Faster ascent when far below target
492 double max_descent = 0.5; // Keep moderate descent rate
493
494 adjustment = std::clamp(adjustment, -max_descent, max_ascent);
495
496 // HARD LIMIT on absolute altitude regardless of terrain
497 if (current_altitude > 8.0)
498 {
499 RCLCPP_ERROR(logger_, "MAXIMUM ALTITUDE REACHED (%.2fm) - EMERGENCY DESCENT", current_altitude);
500 integral_error = 0.0; // Reset integral term
501 return -0.5; // Force immediate descent
502 }
503
504 // Continuous ascent detection (safety feature)
505 static int continuous_ascent_counter = 0;
506 static double last_altitude = 0.0;
507
508 if (current_altitude > last_altitude + 0.05 && adjustment > 0.0)
509 {
510 continuous_ascent_counter++;
511 if (continuous_ascent_counter > 20)
512 {
513 RCLCPP_ERROR(logger_, "CONTINUOUS ASCENT DETECTED (%.2fm) - EMERGENCY CORRECTION",
514 current_altitude);
515 continuous_ascent_counter = 0;
516 integral_error = 0.0; // Reset integral term
517 return -0.5; // Force descent
518 }
519 }
520 else
521 {
522 continuous_ascent_counter = 0;
523 }
524 last_altitude = current_altitude;
525
526 // Add slight damping - prevent oscillation
527 static double last_adjustment = 0.0;
528 double damped_adjustment = 0.7 * adjustment + 0.3 * last_adjustment;
529 last_adjustment = damped_adjustment;
530
531 // More frequent debug logging with integral term info
532 if (debug_counter % 5 == 0)
533 {
534 double ground_elevation = current_altitude - sonar_range;
535 RCLCPP_INFO(logger_,
536 "Altitude control: alt=%.2fm, sonar=%.2fm, target=%.2fm, error=%.2fm, P=%.2f, I=%.2f, cmd=%.2fm/s, ground=%.2fm",
537 current_altitude, sonar_range, target_height, error,
538 p_term, i_term, damped_adjustment, ground_elevation);
539 }
540
541 return damped_adjustment;
542}

554bool Controller::checkObstaclesInDirection(const sensor_msgs::msg::LaserScan::SharedPtr &laser_scan,
555 const geometry_msgs::msg::Point &current_pos,
556 const geometry_msgs::msg::Point &target_pos)
557{
558 if (!laser_scan)
559 {
560 return false;
561 }
562
563 // Calculate direction of travel (NOT drone heading)
564 double dx = target_pos.x - current_pos.x;
565 double dy = target_pos.y - current_pos.y;
566 double travel_distance = std::sqrt(dx * dx + dy * dy);
567
568 if (travel_distance < 0.1)
569 {
570 return false; // Already at target
571 }
572
573 // Normalise direction vector
574 dx /= travel_distance;
575 dy /= travel_distance;
576
577 // Calculate travel direction angle based on navigation
578 double travel_angle = std::atan2(dy, dx);
579
580 // Wide cone to catch obstacles in travel path
581 double cone_half_angle = M_PI / 4.0; // ±45 degrees
582
583 bool obstacle_detected = false;
584 double closest_obstacle = std::numeric_limits<double>::max();
585
586 // Check for obstacles
587 for (size_t i = 0; i < laser_scan->ranges.size(); i++)
588 {
589 float range = laser_scan->ranges[i];
590
591 // Skip invalid readings
592 if (std::isnan(range) || std::isinf(range) ||
593 range <= laser_scan->range_min ||
594 range >= laser_scan->range_max)
595 {
596 continue;
597 }
598
599 // Calculate angle
600 float laser_angle = laser_scan->angle_min + (i * laser_scan->angle_increment);
601
602 // Calculate angle difference from travel direction
603 double angle_diff = std::abs(laser_angle - travel_angle);
604 if (angle_diff > M_PI)
605 {
606 angle_diff = 2 * M_PI - angle_diff;
607 }
608
609 // Check if obstacle is in our travel path
610 if (angle_diff <= cone_half_angle)
611 {
612 // Obstacle in travel direction - check distance
613 if (range < 3.0)
614 { // 3m detection distance
615 obstacle_detected = true;
616 closest_obstacle = std::min(closest_obstacle, static_cast<double>(range));
617
618 RCLCPP_DEBUG(logger_, "Obstacle in travel path: %.2fm at %.1f° (travel dir: %.1f°)",
619 range, laser_angle * 180.0 / M_PI, travel_angle * 180.0 / M_PI);
620 }
621 }
622 }
623
624 if (obstacle_detected)
625 {
626 RCLCPP_WARN(logger_, "OBSTACLE IN TRAVEL PATH: %.2fm - ascending!", closest_obstacle);
627 }
628
629 return obstacle_detected;
630}

632bool Controller::isObstacleClearanceAchieved(const sensor_msgs::msg::LaserScan::SharedPtr &laser_scan,
633 const geometry_msgs::msg::Point &current_pos,
634 const geometry_msgs::msg::Point &target_pos,
635 double clearance_distance)
636{
637 if (!laser_scan)
638 return false;
639
640 // Calculate travel direction
641 double dx = target_pos.x - current_pos.x;
642 double dy = target_pos.y - current_pos.y;
643 double travel_distance = std::sqrt(dx * dx + dy * dy);
644
645 if (travel_distance < 0.1)
646 return true;
647
648 double travel_angle = std::atan2(dy / travel_distance, dx / travel_distance);
649 double cone_half_angle = M_PI / 4.0; // ±45 degrees
650
651 // Check if path is clear
652 for (size_t i = 0; i < laser_scan->ranges.size(); i++)
653 {
654 float range = laser_scan->ranges[i];
655
656 if (std::isnan(range) || std::isinf(range) ||
657 range <= laser_scan->range_min ||
658 range >= laser_scan->range_max)
659 {
660 continue;
661 }
662
663 float laser_angle = laser_scan->angle_min + (i * laser_scan->angle_increment);
664 double angle_diff = std::abs(laser_angle - travel_angle);
665 if (angle_diff > M_PI)
666 angle_diff = 2 * M_PI - angle_diff;
667
668 if (angle_diff <= cone_half_angle && range < clearance_distance)
669 {
670 return false; // Still obstacles in path
671 }
672 }
673
674 return true; // Path is clear
