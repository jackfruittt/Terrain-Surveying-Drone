361 // Define cost function (3D Euclidean distance)
362 auto cost_function = [](const geometry_msgs::msg::Point& from,
363 const geometry_msgs::msg::Point& to) -> double {
364 double dx = to.x - from.x;
365 double dy = to.y - from.y;
366 double dz = to.z - from.z;
367 return std::sqrt(dx*dx + dy*dy + dz*dz);

370 // Define traversability function (drone-appropriate)
371 auto traversability_function = [](const geometry_msgs::msg::Point& from,
372 const geometry_msgs::msg::Point& to) -> bool {
373 // For drone operations, check basic bounds and reasonable altitude changes
374 if (std::abs(from.x) > 20.0 || std::abs(from.y) > 20.0 || from.z < 0 || from.z > 20.0 ||
375 std::abs(to.x) > 20.0 || std::abs(to.y) > 20.0 || to.z < 0 || to.z > 20.0) {
376 return false;
377 }
378
379 // Check for reasonable altitude changes
380 double altitude_diff = std::abs(to.z - from.z);
381 double horizontal_distance = std::sqrt(std::pow(to.x - from.x, 2) + std::pow(to.y - from.y, 2));
382
383 if (horizontal_distance > 0.1) {
384 double climb_angle = std::atan(altitude_diff / horizontal_distance) * 180.0 / M_PI;
385 if (climb_angle > 60.0) {
386 return false;
387 }
388 }
389
390 return true;
391 };
